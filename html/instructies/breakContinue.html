break en continue

break

// break : jumping out
let k = 0;
while (true) {
    k++;
    if (k > 5) 
        break;
    console.log(`Waarde voor k : ${k}`);
}

continue

//skipping with continue
let l = -3
while (l < 3) {
    l++;
    if (l < 0)
        continue;
    console.log(`Waarde voor l : ${l}`);
}
