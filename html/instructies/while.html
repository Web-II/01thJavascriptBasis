while / do while

while

let scoops = 10;
while (scoops > 0) {
   console.log('More icecream!');
   scoops--;
}
console.log("life without ice cream isn't the same");

do while

        
do {
   console.log('More icecream!');
   scoops++;
} while (scoops < 10);
console.log(':)');
