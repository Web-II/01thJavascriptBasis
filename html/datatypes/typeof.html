Datatypes

typeof

De typeof operator retourneert het type. Dit type wordt afgeleid uit de waarde die werd toegekend aan de variabele

let x;  // undefined
typeof x; // undefined
const y = 0; // 0
typeof y; // number
let message = "hallo";
typeof message; // string
typeof 1; // number
typeof 1.0; // number
typeof true; // boolean
typeof 'hello world'; // string
typeof undefined; // undefined
typeof null; // object
const person = {
    name: 'Frank'
};
typeof person; // object
typeof console.log; // function
