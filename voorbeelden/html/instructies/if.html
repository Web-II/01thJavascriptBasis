Instructies

if: als ... dan ...

if

In het geval van 1 instructie mag {} weggelaten worden

let scoops = 2;
if (scoops < 3) {
    console.log('Ice cream is running low!');
}

if ... else ...

let scoops = 4;
if (scoops < 3) {
    console.log('Ice cream is running low!');
}
else if (scoops > 9){
    console.log('Eat faster, the ice cream is going to melt!');    
}

Nog meer keuzes ...

let scoops = 5;
if (scoops == 3) {
    console.log('Ice cream is running low!');
} else if (scoops > 9) {
    console.log('Eat faster, the ice cream is going to melt!');
} else if (scoops == 2) {
    console.log('Going once!');
} else if (scoops == 1) {
    console.log('Going twice!');
} else if (scoops == 0) {
    console.log('Gone!');
} else {
    console.log('Still lots of ice cream left, come and get it.');
}

switch

const theDay = new Date().getDay();
switch (theDay){
    case 5:
        console.log('Finally Friday');
        break;
    case 6:
        console.log('Super Saturday');
        break;
    case 0:
        console.log('Sleepy Sunday');
        break;
    default:
        console.log("I'm really looking forward to this weekend!");
}
